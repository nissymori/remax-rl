\[
\begin{array}{c@{\hspace{3em}}c@{\hspace{3em}}c}
\text{\Large\bfseries Standard RL}
&
&
\text{\Large\bfseries ReMax}
\\[1.0em]
\displaystyle
J_{\mathrm{RL}}(\pi)
=
\mathbb{E}_{A\sim\pi}[\mathcal{R}(a)]
&
\text{\Large\color{softgray}$\Longrightarrow$}
&
\displaystyle
J_{\mathrm{ReMax}}^M(\pi)
=
\mathbb{E}_{\textcolor{softred}{\mathcal{R}\sim\Pi}}
\left[
\mathbb{E}_{\textcolor{softblue}{a_1,\ldots,a_M\sim\pi}}
\left[
\textcolor{softblue}{\max_{m\in[M]}} \mathcal{R}(a_m)
\right]
\right]
\end{array}
